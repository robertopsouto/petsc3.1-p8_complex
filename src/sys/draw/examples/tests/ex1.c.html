  8: int main(int argc,char **argv)
  9: {
 10:   PetscDraw      draw;

 19:   PetscDrawLine(draw,0.0,0.0,1.0,1.0,PETSC_DRAW_BLACK);
 20:   PetscDrawString(draw,.2,.2,PETSC_DRAW_RED,"Some Text");
 21:   PetscDrawStringSetSize(draw,.5,.5);
 22:   PetscDrawString(draw,.2,.2,PETSC_DRAW_BLUE,"Some Text");
 23:   PetscDrawFlush(draw);
 24:   PetscSleep(2);
 25:   PetscDrawClear(draw); PetscDrawFlush(draw);
 26:   PetscDrawResizeWindow(draw,600,600);
 27:   PetscSleep(2);
 28:   PetscDrawLine(draw,0.0,1.0,1.0,0.0,PETSC_DRAW_BLUE);
 29:   PetscDrawFlush(draw);
 30:   PetscSleep(2);
 31:   PetscDrawDestroy(draw);
 32:   PetscFinalize();
