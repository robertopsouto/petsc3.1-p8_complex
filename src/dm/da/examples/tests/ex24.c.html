 12:   DA             da;

 29:       l[j][i][2] = 3*(i + j*M) + 2;
 30:     }
 31:   }
 32:   DAVecRestoreArrayDOF(da,local,&l);
 33:   DALocalToGlobal(da,local,ADD_VALUES,global);

 35:   VecView(global,PETSC_VIEWER_STDOUT_WORLD);

 37:   /* Free memory */
 38:   VecDestroy(local);
 39:   VecDestroy(global);
 40:   DADestroy(da);
